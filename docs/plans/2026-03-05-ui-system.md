# UI System Implementation Plan

**Goal:** Add a 2D UI overlay to game-engine with anchor-based layout, WoW-style widgets, Dioxus component model, and WASM addon plugins.

**Architecture:** The UI is a Bevy plugin (`UiPlugin`) that renders 2D quads on top of the 3D scene. Core data lives in a `UiState` resource (frame registry, anchor solver, event queue, dirty tracking). Dioxus drives the component tree via `WriteMutations` → ECS entity mapping. Addons are `.wasm` modules loaded by wasmtime with a sandboxed `game-api` interface.

**Tech Stack:** Bevy 0.18, dioxus-core 0.7 (VirtualDom + WriteMutations), wasmtime (WASM runtime), image-blp (BLP textures)

```rust
/// 9-point anchor system matching WoW's SetPoint API.
#[derive(Debug, Clone, Copy, PartialEq, Eq, Hash)]
pub enum AnchorPoint {
    TopLeft, Top, TopRight,
    Left, Center, Right,
    BottomLeft, Bottom, BottomRight,
}

impl AnchorPoint {
    pub fn from_str(s: &str) -> Option<Self> {
        match s {
            "TOPLEFT" => Some(Self::TopLeft),
            "TOP" => Some(Self::Top),
            "TOPRIGHT" => Some(Self::TopRight),
            "LEFT" => Some(Self::Left),
            "CENTER" => Some(Self::Center),
            "RIGHT" => Some(Self::Right),
            "BOTTOMLEFT" => Some(Self::BottomLeft),
            "BOTTOM" => Some(Self::Bottom),
            "BOTTOMRIGHT" => Some(Self::BottomRight),
            _ => None,
        }
    }
}

/// An anchor attaching one frame's point to another frame's point.
#[derive(Debug, Clone)]
pub struct Anchor {
    /// Which point on this frame
    pub point: AnchorPoint,
    /// Target frame ID (None = parent)
    pub relative_to: Option<u64>,
    /// Which point on the target frame
    pub relative_point: AnchorPoint,
    /// Pixel offsets from the resolved position
    pub x_offset: f32,
    pub y_offset: f32,
}

/// Get the pixel position of an anchor point on a rect.
/// Rect is (x, y, width, height) in screen coords (Y-down).
pub fn anchor_position(point: AnchorPoint, x: f32, y: f32, w: f32, h: f32) -> (f32, f32) {
    match point {
        AnchorPoint::TopLeft => (x, y),
        AnchorPoint::Top => (x + w / 2.0, y),
        AnchorPoint::TopRight => (x + w, y),
        AnchorPoint::Left => (x, y + h / 2.0),
        AnchorPoint::Center => (x + w / 2.0, y + h / 2.0),
        AnchorPoint::Right => (x + w, y + h / 2.0),
        AnchorPoint::BottomLeft => (x, y + h),
        AnchorPoint::Bottom => (x + w / 2.0, y + h),
        AnchorPoint::BottomRight => (x + w, y + h),
    }
}

/// Get the top-left corner of a frame given which anchor point is at (target_x, target_y).
pub fn frame_position_from_anchor(point: AnchorPoint, target_x: f32, target_y: f32, w: f32, h: f32) -> (f32, f32) {
    match point {
        AnchorPoint::TopLeft => (target_x, target_y),
        AnchorPoint::Top => (target_x - w / 2.0, target_y),
        AnchorPoint::TopRight => (target_x - w, target_y),
        AnchorPoint::Left => (target_x, target_y - h / 2.0),
        AnchorPoint::Center => (target_x - w / 2.0, target_y - h / 2.0),
        AnchorPoint::Right => (target_x - w, target_y - h / 2.0),
        AnchorPoint::BottomLeft => (target_x, target_y - h),
        AnchorPoint::Bottom => (target_x - w / 2.0, target_y - h),
        AnchorPoint::BottomRight => (target_x - w, target_y - h),
    }
}
```

```rust
use crate::ui::anchor::{Anchor, AnchorPoint, anchor_position, frame_position_from_anchor};

#[derive(Debug, Clone, PartialEq)]
pub struct LayoutRect {
    pub x: f32,
    pub y: f32,
    pub width: f32,
    pub height: f32,
}

/// Resolve a frame's position from its anchors, explicit size, and parent rect.
///
/// Single-pass algorithm ported from wow-ui-sim:
/// 1. For each anchor, compute target position on parent (or relative frame)
/// 2. Apply offset (Y inverted: WoW Y-up → screen Y-down)
/// 3. Single anchor: position frame at that point
/// 4. Two anchors: stretch between the two resolved positions
pub fn resolve_anchors(
    anchors: &[Anchor],
    width: f32,
    height: f32,
    parent_rect: &LayoutRect,
) -> LayoutRect {
    // No anchors = top-left of parent
    if anchors.is_empty() {
        return LayoutRect { x: parent_rect.x, y: parent_rect.y, width, height };
    }

    // Resolve first anchor
    let a = &anchors[0];
    let (target_x, target_y) = resolve_one_anchor(a, parent_rect);

    if anchors.len() == 1 {
        let (fx, fy) = frame_position_from_anchor(a.point, target_x, target_y, width, height);
        return LayoutRect { x: fx, y: fy, width, height };
    }

    // Two anchors: resolve both, derive position and possibly stretched size
    let b = &anchors[1];
    let (target_x2, target_y2) = resolve_one_anchor(b, parent_rect);

    // Get the positions that each anchor point implies for the frame's edges
    let (ax, ay) = point_to_edge_offsets(a.point);
    let (bx, by) = point_to_edge_offsets(b.point);

    let mut left = parent_rect.x;
    let mut top = parent_rect.y;
    let mut w = width;
    let mut h = height;

    // If the two anchors define different X positions, stretch horizontally
    if (ax - bx).abs() > f32::EPSILON {
        let x1 = target_x - ax * width;
        let x2 = target_x2 - bx * width;
        left = x1.min(x2);
        w = (x1 - x2).abs() + width * (ax - bx).abs();
        // Recalculate: left is where the frame's left edge lands
        left = if ax < bx { target_x } else { target_x2 };
        let right = if ax < bx { target_x2 } else { target_x };
        w = right - left;
        // adjust left back from the anchor offset
    }

    // If the two anchors define different Y positions, stretch vertically
    if (ay - by).abs() > f32::EPSILON {
        let top_target = if ay < by { target_y } else { target_y2 };
        let bottom_target = if ay < by { target_y2 } else { target_y };
        top = top_target;
        h = bottom_target - top_target;
    }

    // For same-axis anchors (e.g., both Left), use first anchor for positioning
    if (ax - bx).abs() <= f32::EPSILON {
        let (fx, _) = frame_position_from_anchor(a.point, target_x, target_y, w, h);
        left = fx;
    }
    if (ay - by).abs() <= f32::EPSILON {
        let (_, fy) = frame_position_from_anchor(a.point, target_x, target_y, w, h);
        top = fy;
    }

    LayoutRect { x: left, y: top, width: w, height: h }
}

fn resolve_one_anchor(anchor: &Anchor, parent_rect: &LayoutRect) -> (f32, f32) {
    // TODO: when relative_to is Some, look up that frame's rect instead
    let (ax, ay) = anchor_position(
        anchor.relative_point,
        parent_rect.x, parent_rect.y,
        parent_rect.width, parent_rect.height,
    );
    // Y-offset inverted: WoW uses Y-up, screen uses Y-down
    (ax + anchor.x_offset, ay - anchor.y_offset)
}

/// Returns (x_frac, y_frac) where 0.0 = left/top edge, 0.5 = center, 1.0 = right/bottom edge
fn point_to_edge_offsets(point: AnchorPoint) -> (f32, f32) {
    match point {
        AnchorPoint::TopLeft => (0.0, 0.0),
        AnchorPoint::Top => (0.5, 0.0),
        AnchorPoint::TopRight => (1.0, 0.0),
        AnchorPoint::Left => (0.0, 0.5),
        AnchorPoint::Center => (0.5, 0.5),
        AnchorPoint::Right => (1.0, 0.5),
        AnchorPoint::BottomLeft => (0.0, 1.0),
        AnchorPoint::Bottom => (0.5, 1.0),
        AnchorPoint::BottomRight => (1.0, 1.0),
    }
}
```

```rust
use std::collections::{HashMap, HashSet};
use crate::ui::frame::{Frame, WidgetType};
use crate::ui::layout::LayoutRect;

pub struct FrameRegistry {
    frames: HashMap<u64, Frame>,
    names: HashMap<String, u64>,
    next_id: u64,
    screen_width: f32,
    screen_height: f32,
    render_dirty: HashSet<u64>,
    rect_dirty: HashSet<u64>,
    /// Reverse index: target_id → set of frame IDs anchored to it
    anchor_dependents: HashMap<u64, HashSet<u64>>,
}

impl FrameRegistry {
    pub fn new(screen_width: f32, screen_height: f32) -> Self {
        Self {
            frames: HashMap::new(),
            names: HashMap::new(),
            next_id: 1,
            screen_width,
            screen_height,
            render_dirty: HashSet::new(),
            rect_dirty: HashSet::new(),
            anchor_dependents: HashMap::new(),
        }
    }

    pub fn screen_rect(&self) -> LayoutRect {
        LayoutRect { x: 0.0, y: 0.0, width: self.screen_width, height: self.screen_height }
    }

    pub fn create_frame(&mut self, name: &str, parent_id: Option<u64>) -> u64 {
        let id = self.next_id;
        self.next_id += 1;

        let mut frame = Frame::new(id, Some(name.to_string()), WidgetType::Frame);
        frame.parent_id = parent_id;

        // Inherit effective values from parent
        if let Some(pid) = parent_id {
            if let Some(parent) = self.frames.get(&pid) {
                frame.effective_alpha = parent.effective_alpha * frame.alpha;
                frame.effective_scale = parent.effective_scale * frame.scale;
                frame.visible = parent.visible && frame.shown;
                frame.frame_level = parent.frame_level + 1;
            }
        }

        if let Some(ref n) = frame.name {
            self.names.insert(n.clone(), id);
        }

        self.frames.insert(id, frame);

        // Add to parent's children list
        if let Some(pid) = parent_id {
            if let Some(parent) = self.frames.get_mut(&pid) {
                parent.children.push(id);
            }
        }

        self.render_dirty.insert(id);
        id
    }

    pub fn get(&self, id: u64) -> Option<&Frame> {
        self.frames.get(&id)
    }

    pub fn get_mut(&mut self, id: u64) -> Option<&mut Frame> {
        self.render_dirty.insert(id);
        self.frames.get_mut(&id)
    }

    pub fn get_by_name(&self, name: &str) -> Option<u64> {
        self.names.get(name).copied()
    }

    pub fn set_alpha(&mut self, id: u64, alpha: f32) {
        if let Some(frame) = self.frames.get_mut(&id) {
            frame.alpha = alpha;
        }
        self.propagate_alpha(id);
    }

    pub fn set_shown(&mut self, id: u64, shown: bool) {
        if let Some(frame) = self.frames.get_mut(&id) {
            frame.shown = shown;
        }
        self.propagate_visibility(id);
    }

    fn propagate_alpha(&mut self, id: u64) {
        let parent_eff = self.frames.get(&id)
            .and_then(|f| f.parent_id)
            .and_then(|pid| self.frames.get(&pid))
            .map(|p| p.effective_alpha)
            .unwrap_or(1.0);

        if let Some(frame) = self.frames.get_mut(&id) {
            let new_eff = if frame.visible { parent_eff * frame.alpha } else { 0.0 };
            frame.effective_alpha = new_eff;
            self.render_dirty.insert(id);
        }

        let children: Vec<u64> = self.frames.get(&id)
            .map(|f| f.children.clone())
            .unwrap_or_default();

        for child_id in children {
            self.propagate_alpha(child_id);
        }
    }

    fn propagate_visibility(&mut self, id: u64) {
        let parent_visible = self.frames.get(&id)
            .and_then(|f| f.parent_id)
            .and_then(|pid| self.frames.get(&pid))
            .map(|p| p.visible)
            .unwrap_or(true);

        if let Some(frame) = self.frames.get_mut(&id) {
            frame.visible = parent_visible && frame.shown;
            self.render_dirty.insert(id);
        }

        // Also update alpha (visibility affects effective alpha)
        self.propagate_alpha(id);

        let children: Vec<u64> = self.frames.get(&id)
            .map(|f| f.children.clone())
            .unwrap_or_default();

        for child_id in children {
            self.propagate_visibility(child_id);
        }
    }
}
```

```rust
    OnAnimFinished, OnAnimLoop, OnAnimPlay, OnAnimStop,
    OnPostShow, OnPostHide, OnPostUpdate,

/// Event bus: maps event names → registered frame IDs, plus a pending event queue.
pub struct EventBus {
    /// event_name → set of frame IDs listening
    registrations: HashMap<String, HashSet<u64>>,
    /// Pending events to dispatch
    queue: Vec<UiEvent>,
}

impl EventBus {
    pub fn new() -> Self {
        Self { registrations: HashMap::new(), queue: Vec::new() }
    }

    pub fn register(&mut self, frame_id: u64, event: &str) {
        self.registrations.entry(event.to_string()).or_default().insert(frame_id);
    }

    pub fn unregister(&mut self, frame_id: u64, event: &str) {
        if let Some(set) = self.registrations.get_mut(event) {
            set.remove(&frame_id);
        }
    }

    pub fn unregister_all(&mut self, frame_id: u64) {
        for set in self.registrations.values_mut() {
            set.remove(&frame_id);
        }
    }

    pub fn listeners(&self, event: &str) -> Vec<u64> {
        self.registrations.get(event)
            .map(|s| { let mut v: Vec<_> = s.iter().copied().collect(); v.sort(); v })
            .unwrap_or_default()
    }

    pub fn push(&mut self, event: UiEvent) {
        self.queue.push(event);
    }

    pub fn drain(&mut self) -> Vec<UiEvent> {
        std::mem::take(&mut self.queue)
    }
}
```
